import { 
  Heart, 
  Share2, 
  MessageCircle, 
  ArrowUp, 
  Play, 
  Volume2, 
  VolumeX,
  MapPin,
  Calendar,
  CreditCard,
  Info

  const [isMuted, setIsMuted] = useState(true);
  const [liked, setLiked] = useState(false);
  const containerRef = useRef(null);

  // Data Per "Slide"
  const slides = [
    {
      type: 'hero',
      title: "AMALFI COAST",

    {
      day: 1,
      location: "Positano",
      title: "Vertical City Vibes",
      desc: "Tiba di Naples dan transfer ke Positano. Jelajahi gang-gang sempit penuh bunga bougainvillea.",

  useEffect(() => {

      const scrollPos = containerRef.current.scrollTop;
      const height = window.innerHeight;
      const index = Math.round(scrollPos / height);
      setActiveScreen(index);
    };

        @import url('https://fonts.googleapis.com/css2?family=Montserrat:wght@400;700;900&display=swap');

                <div className="mb-24 animate-in fade-in slide-in-from-bottom-8 duration-1000">
                  <span className="bg-white text-black px-3 py-1 rounded text-[10px] font-black uppercase tracking-tighter mb-4 inline-block">Exclusive Tour</span>
                  <h1 className="text-6xl md:text-8xl font-black italic tracking-tighter leading-none mb-2">{slide.title}</h1>
                  <p className="text-xl font-light tracking-widest opacity-80 uppercase">{slide.subtitle}</p>
                  <div className="flex gap-4 mt-6">
                    <div className="flex flex-col">
                      <span className="text-[10px] opacity-50 uppercase font-bold">Investasi</span>
                      <span className="text-2xl font-bold">{slide.price}</span>
                    </div>
                    <div className="w-[1px] bg-white/30 h-10"></div>
                    <div className="flex flex-col">
                      <span className="text-[10px] opacity-50 uppercase font-bold">Periode</span>
                      <span className="text-lg font-bold">{slide.period}</span>
                    </div>
                  </div>
                </div>
              )}

              {slide.day && (
                <div className="mb-32 max-w-lg">
                  <div className="flex items-center gap-2 text-white/70 text-xs mb-2">
                    <MapPin size={14} /> {slide.location}
                  </div>
                  <h2 className="text-4xl font-black uppercase mb-4">Day {slide.day}: {slide.title}</h2>
                  <p className="text-white/80 leading-relaxed font-light">{slide.desc}</p>
                  <button className="mt-6 glass-ui px-6 py-2 rounded-full text-xs font-bold uppercase tracking-widest flex items-center gap-2">
                    <Play size={12} fill="white" /> Lihat Visual
                  </button>
                </div>
              )}

              {slide.type === 'deck' && (
                <div className="mb-32 grid grid-cols-1 gap-3">
                   <h2 className="text-3xl font-black mb-4 italic">READY FOR TAKE OFF?</h2>
                   <div className="glass-ui p-6 rounded-3xl space-y-4">
                      <div className="flex justify-between items-center border-b border-white/10 pb-2">
                         <span className="flex items-center gap-2 text-xs uppercase font-bold tracking-widest"><Calendar size={14}/> Keberangkatan</span>
                         <span className="text-sm">24 Agustus 2024</span>
                      </div>
                      <div className="flex justify-between items-center border-b border-white/10 pb-2">
                         <span className="flex items-center gap-2 text-xs uppercase font-bold tracking-widest"><CreditCard size={14}/> Harga Include</span>
                         <span className="text-sm underline cursor-pointer">Lihat Detail</span>
                      </div>
                      <div className="flex justify-between items-center">
                         <span className="flex items-center gap-2 text-xs uppercase font-bold tracking-widest"><Info size={14}/> S&K</span>
                         <span className="text-sm">Swipe for Info</span>
                      </div>
                   </div>
                </div>
              )}
            </div>
          </section>
        ))}

      {/* --- RIGHT SIDEBAR ACTIONS (TikTok Style) --- */}

          className="p-3 rounded-full glass-ui mt-4"

            key={i} 
            className={`w-1 transition-all duration-300 rounded-full ${activeScreen === i ? 'h-8 bg-white' : 'h-3 bg-white/30'}`}
          />
        ))}
      </div>

      {/* --- BOTTOM CTA --- */}
      <div className={`fixed bottom-0 left-0 right-0 p-6 bg-gradient-to-t from-black to-transparent transition-all duration-500 transform ${activeScreen > 0 ? 'translate-y-0' : 'translate-y-full'}`}>
