    <script>
        // GALLERY IMAGES
        const galleryImages = [
            'https://images.unsplash.com/photo-1493976040374-85c8e12f0c0e?w=1200',
            'https://images.unsplash.com/photo-1540959733332-eab4deabeeaf?w=1200',
            'https://images.unsplash.com/photo-1578469550956-0e16b69c6a3d?w=1200'
        ];
        
        // LIGHTBOX
        function openLightbox(index) {
            const lightbox = document.getElementById('lightbox');
            const img = document.getElementById('lightboxImg');
            img.src = galleryImages[index];
            lightbox.classList.add('active');
            document.body.style.overflow = 'hidden';
        }
        
        function closeLightbox() {
            const lightbox = document.getElementById('lightbox');
            lightbox.classList.remove('active');
            document.body.style.overflow = '';
        }
        
        // COLLAPSIBLE DAYS
        function toggleDay(element) {
            const isActive = element.classList.contains('active');
            
            // Close all
            document.querySelectorAll('.day-item').forEach(day => {
                day.classList.remove('active');
            });
            
            // Open clicked if wasn't active
            if (!isActive) {
                element.classList.add('active');
            }
        }
        
        // OPEN FIRST DAY BY DEFAULT
        document.querySelector('.day-item').classList.add('active');
        
        // SMOOTH SCROLL FOR NAV
        document.querySelectorAll('.nav-pill').forEach(pill => {
            pill.addEventListener('click', (e) => {
                e.preventDefault();
                const target = document.querySelector(pill.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({ behavior: 'smooth', block: 'start' });
                    
                    // Update active state
                    document.querySelectorAll('.nav-pill').forEach(p => p.classList.remove('active'));
                    pill.classList.add('active');
                }
            });
        });
        
        // UPDATE NAV ON SCROLL
        const sections = document.querySelectorAll('.section, .gallery');
        const navPills = document.querySelectorAll('.nav-pill');
        
        window.addEventListener('scroll', () => {
            let current = '';
            sections.forEach(section => {
                const sectionTop = section.offsetTop;
                if (scrollY >= sectionTop - 200) {
                    current = section.getAttribute('id');
                }
            });
            
            navPills.forEach(pill => {
                pill.classList.remove('active');
                if (pill.getAttribute('href') === '#' + current) {
                    pill.classList.add('active');
                }
            });
        });
    </script>
